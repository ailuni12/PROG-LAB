	min,seg,cent, segAux, seg1, seg2, seg3: R
	nomAux, nom1, nom2, nom3: C[30]

Comienzo
	
	Escribir "Ingrese nombre del primer atleta: "
	Leer nom1
	Escribir "Ingrese su tiempo en minutos, segundos y centisegundos: "
	Leer min, seg, cent
	seg1 <- min*60 + seg + cent/100
	
	Escribir "Ingrese nombre del segundo atleta: "
	Leer nomAux
	Escribir "Ingrese su tiempo en minutos, segundos y centisegundos: "
	Leer min, seg, cent
	segAux <- min*60 + seg + cent/100
	
	Si segAux<seg1 Entonces
		seg2<-seg1
		nom2<-nom1
		seg1<-segAux
		nom1<-nomAux
	SiNo
		seg2<-segAux
		nom2<-nomAux
	Fin Si
	
	Escribir "Ingrese nombre del tercer atleta: "
	Leer nomAux
	Escribir "Ingrese su tiempo en minutos, segundos y centisegundos: "
	Leer min, seg, cent
	segAux <- min*60 + seg + cent/100
	
	Si segAux<seg1 Entonces
		seg3<-seg2
		nom3<-nom2
		seg2<-seg1
		nom2<-nom1
		seg1<-segAux
		nom1<-nomAux
	SiNo
		Si segAux<seg2 Entonces
			seg3<-seg2
			nom3<-nom2
			seg2<-segAux
			nom2<-nomAux
		SiNo
			seg3<-segAux
			nom3<-nomAux
		FinSi
	Fin Si
	
	Escribir "Ingrese nombre del cuarto atleta: "
	Leer nomAux
	Escribir "Ingrese su tiempo en minutos, segundos y centisegundos: "
	Leer min, seg, cent
	segAux <- min*60 + seg + cent/100
	
	Si segAux<seg1 Entonces
		seg3<-seg2
		nom3<-nom2
		seg2<-seg1
		nom2<-nom1
		seg1<-segAux
		nom1<-nomAux
	SiNo
		Si segAux<seg2 Entonces
			seg3<-seg2
			nom3<-nom2
			seg2<-segAux
			nom2<-nomAux
		SiNo
			Si segAux<seg3 Entonces
				seg3<-segAux
				nom3<-nomAux
			FinSi
		FinSi
	Fin Si
	
	Escribir "1º puesto: ", nom1, " con: ", seg1, " segundos."
	Escribir "2º puesto: ", nom2, " con: ", seg2, " segundos."
	Escribir "3º puesto: ", nom3, " con: ", seg3, " segundos."
